% G : [R&, R&, T&] -> C&
\[ G : [\mathcal R, \mathcal R, \mathcal T] \to \mathcal C \]

% G[r#, r#', t] =- 1/(i hbar) <r#| exp (H t)/(i h-)
\[ G[\vc r, \vc r', t]
\equiv
\frac{1}{\I \hbar}
\brakket{\vc r}{\exp \frac{\op H t}{\I \hbar}}{\vc r'}
=
\frac{1}{\I \hbar}
\sum_{n} (\phi_n \vc r) \exp \frac{E_n t}{\I \hbar} (\phi_n^* \vc r')
\]
